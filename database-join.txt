SELECT YEAR(`enrolment_date`), COUNT(id) 
FROM `students` 
GROUP BY YEAR(`enrolment_date`); 

SELECT COUNT(id), office_address 
FROM `teachers` 
GROUP BY `office_address`; 

SELECT exam_id, AVG(vote) 
FROM exam_student 
GROUP BY exam_id

SELECT department_id, COUNT(id) 
FROM `degrees`
GROUP BY (department_id);



FINE GROUP BY
SELECT CONCAT(students.name, students.surname) as fullname, degrees.name 
FROM `students` 
JOIN degrees 
ON students.degree_id = degrees.id 
WHERE degrees.name LIKE 'Corso di Laurea in Economia';

SELECT departments.name, degrees.level
FROM `departments`
JOIN degrees
ON departments.id = degrees.department_id
WHERE degrees.level LIKE 'magistrale' AND departments.name LIKE 'Dipartimento di Neuroscienze'; 

SELECT teachers.name, teachers.surname, courses.*
FROM `teachers`
JOIN course_teacher
ON teachers.id = course_teacher.teacher_id
JOIN courses
ON course_teacher.teacher_id = courses.id
WHERE teachers.id = 44;
